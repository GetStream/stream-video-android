The audio indicator gives you a visual feedback when a user is speaking.
To understand who's speaking we provide call.state.dominantSpeaker and call.state.activeSpeakers.
On the participant you can observe participant.speaking, participant.dominantSpeaker and participant.audioLevel

This example shows how to render a visual UI indicator that changes based on the audio level

```kotlin
val audioLevel = participant.audioLevel.collectAsState()

@Composable
fun AudioVolumeIndicator() {
    val audioViewModel: AudioViewModel = viewModel()
    val audioVolume by audioViewModel.audioVolume.collectAsState()
    val totalDots = 5
    val activeDots = (audioVolume * totalDots).toInt()

    Column {
        Icon(
            painter = painterResource(R.drawable.ic_speaker), // Replace with your speaker icon
            contentDescription = null,
            modifier = Modifier.size(24.dp)
        )
        for (i in 1..totalDots) {
            Box(
                modifier = Modifier.size(8.dp),
                shape = CircleShape
            ) {
                Surface(
                    shape = CircleShape,
                    color = if (i <= activeDots) MaterialTheme.colors.primary else Color.Gray
                ) {}
            }
        }
    }
}
```
